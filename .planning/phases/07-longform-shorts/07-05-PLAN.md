### Task 3: Longform intake page (URL + file upload)

**File**: `src/app/(dashboard)/longform/new/page.tsx` (NEW)

**Action**: client-rendered form with two tabs: URL and File Upload. URL tab calls `POST /api/longform/sources` directly. File tab: (1) call `POST /api/longform/sources/upload-url` to get a signed upload URL, (2) use `supabase.storage.from('longform-sources').uploadToSignedUrl(path, token, file)` in the browser, (3) call `POST /api/longform/sources` with `{ sourceType: 'file', storagePath }`. After either flow, `router.push(\`/longform/\${sourceId}\`)`.

```tsx
"use client";

import { useState } from "react";
import { useRouter } from "next/navigation";
import { createBrowserClient } from "@supabase/ssr";
import { Tabs, TabsList, TabsTrigger, TabsContent } from "@/components/ui/tabs";
import { Input } from "@/components/ui/input";
import { Button } from "@/components/ui/button";
import { Progress } from "@/components/ui/progress";
import { toast } from "sonner";

export default function LongformNewPage() {
  const router = useRouter();
  const [url, setUrl] = useState("");
  const [file, setFile] = useState<File | null>(null);
  const [uploading, setUploading] = useState(false);
  const [uploadPct, setUploadPct] = useState(0);

  async function submitUrl() {
    const res = await fetch("/api/longform/sources", {
      method: "POST",
      headers: { "content-type": "application/json" },
      body: JSON.stringify({ sourceType: "url", url }),
    });
    if (!res.ok) { toast.error((await res.json()).error ?? "실패"); return; }
    const { sourceId } = await res.json();
    router.push(`/longform/${sourceId}`);
  }

  async function submitFile() {
    if (!file) return;
    setUploading(true);
    try {
      const signed = await fetch("/api/longform/sources/upload-url", {
        method: "POST",
        headers: { "content-type": "application/json" },
        body: JSON.stringify({ filename: file.name, contentType: file.type, fileSize: file.size }),
      }).then((r) => r.json());
      if (signed.error) throw new Error(signed.error);

      const supabase = createBrowserClient(
        process.env.NEXT_PUBLIC_SUPABASE_URL!,
        process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!
      );
      const { error: uploadError } = await supabase.storage
        .from("longform-sources")
        .uploadToSignedUrl(signed.storagePath, signed.token, file, {
          contentType: file.type,
          upsert: true,
        });
      if (uploadError) throw uploadError;
      setUploadPct(100);

      const create = await fetch("/api/longform/sources", {
        method: "POST",
        headers: { "content-type": "application/json" },
        body: JSON.stringify({ sourceType: "file", storagePath: signed.storagePath, title: file.name }),
      }).then((r) => r.json());
      if (create.error) throw new Error(create.error);

      router.push(`/longform/${create.sourceId}`);
    } catch (err) {
      toast.error((err as Error).message);
    } finally {
      setUploading(false);
    }
  }

  return (
    <div className="mx-auto max-w-2xl space-y-6 p-6">
      <h1 className="text-2xl font-semibold">새 롱폼 → 쇼츠</h1>
      <Tabs defaultValue="url">
        <TabsList className="grid w-full grid-cols-2">
          <TabsTrigger value="url">YouTube URL</TabsTrigger>
          <TabsTrigger value="file">파일 업로드</TabsTrigger>
        </TabsList>
        <TabsContent value="url" className="space-y-4 pt-4">
          <Input placeholder="https://www.youtube.com/watch?v=..." value={url} onChange={(e) => setUrl(e.target.value)} />
          <Button onClick={submitUrl} disabled={!url}>다운로드 시작</Button>
        </TabsContent>
        <TabsContent value="file" className="space-y-4 pt-4">
          <Input
            type="file"
            accept="video/mp4,video/quicktime,video/webm,video/x-matroska"
            onChange={(e) => setFile(e.target.files?.[0] ?? null)}
          />
          {uploading && <Progress value={uploadPct} />}
          <Button onClick={submitFile} disabled={!file || uploading}>업로드 시작</Button>
        </TabsContent>
      </Tabs>
    </div>
  );
}
```

### Task 5: LongformDetailClient — orchestrates progress + candidate grid

**File**: `src/components/longform/longform-detail-client.tsx` (NEW)

**Action**: client component responsibilities:
1. Poll `/api/longform/sources/[id]` every 2 seconds while `source.status in ('pending','downloading','analyzing')`
2. After `source.status === 'ready'`, trigger `POST /api/longform/sources/[id]/analyze` once (idempotent: skip if `candidates.length > 0` or status is already `analyzing|analyzed`)
3. When `candidates.length > 0`, render `<CandidateGrid />`
4. When user clicks "Clip selected" or "Auto-clip all", POST to `/api/longform/candidates/clip` and show a progress toast
5. Render a child project list at the bottom for candidates where `selected=true`

```tsx
"use client";

import { useEffect, useMemo, useState } from "react";
import { useRouter } from "next/navigation";
import { toast } from "sonner";
import { Button } from "@/components/ui/button";
import { Badge } from "@/components/ui/badge";
import { CandidateGrid, type Candidate } from "./candidate-grid";
import { LongformProgress } from "./longform-progress";

type Source = {
  id: string;
  title: string | null;
  durationSeconds: number | null;
  status: "pending" | "downloading" | "analyzing" | "ready" | "analyzed" | "failed";
  errorMessage: string | null;
  publicUrl: string | null;
  sourceType: "url" | "file";
};

export function LongformDetailClient(props: {
  sourceId: string;
  initialSource: Source;
  initialCandidates: Candidate[];
}) {
  const router = useRouter();
  const [source, setSource] = useState<Source>(props.initialSource);
  const [candidates, setCandidates] = useState<Candidate[]>(props.initialCandidates);
  const [selected, setSelected] = useState<Set<string>>(new Set());
  const [analyzing, setAnalyzing] = useState(false);
  const [clipping, setClipping] = useState(false);

  // Poll status
  useEffect(() => {
    if (["analyzed", "failed"].includes(source.status) && candidates.length > 0) return;

    const interval = setInterval(async () => {
      const res = await fetch(`/api/longform/sources/${props.sourceId}`, { cache: "no-store" });
      if (!res.ok) return;
      const { source: s, candidates: c } = await res.json();
      setSource(s);
      setCandidates(c);
    }, 2000);
    return () => clearInterval(interval);
  }, [source.status, candidates.length, props.sourceId]);

  // Auto-kick analyze when ready
  useEffect(() => {
    if (source.status === "ready" && candidates.length === 0 && !analyzing) {
      setAnalyzing(true);
      fetch(`/api/longform/sources/${props.sourceId}/analyze`, {
        method: "POST",
        headers: { "content-type": "application/json" },
        body: JSON.stringify({}),
      }).catch((err) => toast.error(err.message)).finally(() => setAnalyzing(false));
    }
  }, [source.status, candidates.length, analyzing, props.sourceId]);

  const toggleSelect = (id: string) => {
    setSelected((prev) => {
      const next = new Set(prev);
      if (next.has(id)) next.delete(id);
      else next.add(id);
      return next;
    });
  };

  async function clipSelected(mode: "selected" | "all") {
    setClipping(true);
    try {
      const body = mode === "all"
        ? { mode: "all" as const, sourceId: props.sourceId }
        : { mode: "selected" as const, sourceId: props.sourceId, candidateIds: Array.from(selected) };
      const res = await fetch(`/api/longform/candidates/clip`, {
        method: "POST",
        headers: { "content-type": "application/json" },
        body: JSON.stringify(body),
      });
      const data = await res.json();
      if (!res.ok) throw new Error(data.error);
      toast.success(`${data.count}개 클립 작업 큐에 등록됨`);
      router.refresh();
    } catch (err) {
      toast.error((err as Error).message);
    } finally {
      setClipping(false);
    }
  }

  return (
    <div className="space-y-6 p-6">
      <header className="space-y-2">
        <h1 className="text-2xl font-semibold">{source.title ?? "롱폼 소스"}</h1>
        <div className="flex items-center gap-3 text-sm text-muted-foreground">
          <Badge>{source.status}</Badge>
          {source.durationSeconds && <span>{Math.round(source.durationSeconds / 60)}분</span>}
        </div>
        {source.errorMessage && <p className="text-sm text-red-500">{source.errorMessage}</p>}
      </header>

      {["pending", "downloading", "analyzing"].includes(source.status) && (
        <LongformProgress sourceId={props.sourceId} />
      )}

      {candidates.length > 0 && (
        <section className="space-y-4">
          <div className="flex items-center justify-between">
            <h2 className="text-lg font-medium">{candidates.length}개 후보</h2>
            <div className="flex gap-2">
              <Button variant="outline" onClick={() => clipSelected("selected")} disabled={selected.size === 0 || clipping}>
                선택한 {selected.size}개 클립
              </Button>
              <Button onClick={() => clipSelected("all")} disabled={clipping}>
                전체 자동 클립
              </Button>
            </div>
          </div>
          <CandidateGrid candidates={candidates} selectedIds={selected} onToggle={toggleSelect} />
        </section>
      )}
    </div>
  );
}
```

```tsx
"use client";

import Link from "next/link";
import { Checkbox } from "@/components/ui/checkbox";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Button } from "@/components/ui/button";

export type Candidate = {
  id: string;
  startMs: number;
  endMs: number;
  hookScore: number;
  emotionalScore: number;
  informationDensity: number;
  trendScore: number;
  reason: string;
  titleSuggestion: string | null;
  transcriptSnippet: string | null;
  selected: boolean;
  childProjectId: string | null;
};

function fmt(ms: number) {
  const s = Math.round(ms / 1000);
  return `${Math.floor(s / 60)}:${(s % 60).toString().padStart(2, "0")}`;
}

function ScoreBar({ label, value }: { label: string; value: number }) {
  return (
    <div className="space-y-1">
      <div className="flex justify-between text-xs text-muted-foreground">
        <span>{label}</span>
        <span>{value}</span>
      </div>
      <div className="h-1.5 w-full overflow-hidden rounded-full bg-muted">
        <div className="h-full bg-primary transition-all" style={{ width: `${value}%` }} />
      </div>
    </div>
  );
}

export function CandidateGrid(props: {
  candidates: Candidate[];
  selectedIds: Set<string>;
  onToggle: (id: string) => void;
}) {
  return (
    <div className="grid grid-cols-1 gap-4 md:grid-cols-2 lg:grid-cols-3">
      {props.candidates.map((c) => (
        <Card key={c.id} className="relative">
          <CardHeader>
            <div className="flex items-start justify-between gap-2">
              <div className="space-y-1">
                <div className="text-xs text-muted-foreground">{fmt(c.startMs)} → {fmt(c.endMs)}</div>
                <CardTitle className="text-base leading-tight">{c.titleSuggestion ?? "(제목 없음)"}</CardTitle>
              </div>
              <Checkbox
                checked={props.selectedIds.has(c.id) || c.selected}
                disabled={c.selected}
                onCheckedChange={() => props.onToggle(c.id)}
              />
            </div>
          </CardHeader>
          <CardContent className="space-y-3">
            <ScoreBar label="Hook" value={c.hookScore} />
            <ScoreBar label="Emotion" value={c.emotionalScore} />
            <ScoreBar label="Density" value={c.informationDensity} />
            <ScoreBar label="Trend" value={c.trendScore} />
            <p className="line-clamp-3 text-sm text-muted-foreground" title={c.reason}>{c.reason}</p>
            {c.childProjectId && (
              <Button asChild size="sm" variant="outline" className="w-full">
                <Link href={`/projects/${c.childProjectId}`}>프로젝트 열기</Link>
              </Button>
            )}
          </CardContent>
        </Card>
      ))}
    </div>
  );
}
```

```tsx
"use client";
import { Progress } from "@/components/ui/progress";
import { useEffect, useState } from "react";

export function LongformProgress({ sourceId }: { sourceId: string }) {
  const [state, setState] = useState<{ progress: number; currentStep: string | null; status: string } | null>(null);
  useEffect(() => {
    const tick = async () => {
      const res = await fetch(`/api/longform/sources/${sourceId}`, { cache: "no-store" });
      if (!res.ok) return;
      const { latestJob } = await res.json();
      if (latestJob) setState({ progress: latestJob.progress, currentStep: latestJob.currentStep, status: latestJob.status });
    };
    tick();
    const interval = setInterval(tick, 1500);
    return () => clearInterval(interval);
  }, [sourceId]);

  if (!state) return <div className="text-sm text-muted-foreground">작업 초기화 중...</div>;
  return (
    <div className="space-y-2">
      <div className="flex justify-between text-sm">
        <span>{state.currentStep ?? state.status}</span>
        <span>{state.progress}%</span>
      </div>
      <Progress value={state.progress} />
    </div>
  );
}
```

## Risks & Mitigations

- **Risk**: Supabase `uploadToSignedUrl` browser SDK chokes on files near 2 GB. **Mitigation**: Supabase Storage supports resumable uploads natively for signed uploads; if issues appear, fall back to `tus-js-client` or the Upload API's resumable protocol. Noted as a known risk in CONTEXT.md.
- **Risk**: Polling load when many users watch the same longform page. **Mitigation**: 2s interval with `cache: "no-store"` is cheap; upgrade to Supabase Realtime subscription in a polish pass.
- **Risk**: Multiple concurrent page instances triggering `POST analyze` twice. **Mitigation**: the analyze endpoint returns 409 if status is not `ready`; the client component also guards with `analyzing` boolean + `candidates.length > 0` check.

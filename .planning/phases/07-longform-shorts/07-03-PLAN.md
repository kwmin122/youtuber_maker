## Goal

Implement the `longform-analyze` job so that given a `longform_sources.id` with `status='ready'`, the worker produces between `minCandidates` and `maxCandidates` (default 5-10) rows in `longform_candidates` with four viral scores, a reason, a title suggestion, and transcript snippet. Both transcript mode (reuse `youtube-transcript`) and audio mode (Gemini Files API) must work.

### Task 2: Gemini Files API + 2.5 Pro integration

**File**: `src/lib/ai/gemini.ts` (MODIFY)

**Action**: extend `createGeminiProvider` to implement the two new methods. Use `genAI.getGenerativeModel({ model })` with a custom model name, and use `@google/generative-ai`'s `GoogleAIFileManager` (or the REST Files API via `fetch`) to upload audio. The `@google/generative-ai` package exposes `new GoogleAIFileManager(apiKey)` from `@google/generative-ai/server`.

```ts
import { GoogleGenerativeAI } from "@google/generative-ai";
import { GoogleAIFileManager } from "@google/generative-ai/server";
import type { AIProvider, GenerateOptions } from "./types";

export function createGeminiProvider(apiKey: string): AIProvider {
  const genAI = new GoogleGenerativeAI(apiKey);
  const fileManager = new GoogleAIFileManager(apiKey);

  return {
    name: "gemini",

    async generateText(prompt, options) {
      return generateTextInternal(genAI, prompt, { ...options, model: "gemini-2.0-flash" });
    },

    async generateTextWithModel(prompt, options) {
      return generateTextInternal(genAI, prompt, options);
    },

    async generateJsonFromAudio(params) {
      const uploadResp = await fileManager.uploadFile(params.audioPath, {
        mimeType: params.mimeType,
        displayName: `longform-audio-${Date.now()}`,
      });
      const fileUri = uploadResp.file.uri;

      // Wait for file to become ACTIVE (large audio files need a few seconds)
      let file = await fileManager.getFile(uploadResp.file.name);
      const startWait = Date.now();
      while (file.state === "PROCESSING" && Date.now() - startWait < 120_000) {
        await new Promise((r) => setTimeout(r, 2000));
        file = await fileManager.getFile(uploadResp.file.name);
      }
      if (file.state !== "ACTIVE") {
        throw new Error(`Gemini file upload failed: state=${file.state}`);
      }

      const model = genAI.getGenerativeModel({
        model: params.model ?? "gemini-2.5-pro",
        generationConfig: { responseMimeType: "application/json", temperature: 0.3 },
        ...(params.systemInstruction ? { systemInstruction: params.systemInstruction } : {}),
      });

      const result = await model.generateContent([
        { fileData: { fileUri, mimeType: params.mimeType } },
        { text: params.prompt },
      ]);

      const text = result.response.text();
      // Delete file server-side (best-effort)
      await fileManager.deleteFile(uploadResp.file.name).catch(() => {});

      if (!text) throw new Error("Gemini audio analysis returned empty response");
      return text;
    },
  };
}

async function generateTextInternal(
  genAI: GoogleGenerativeAI,
  prompt: string,
  options: GenerateOptions & { model: string }
): Promise<string> {
  const model = genAI.getGenerativeModel({
    model: options.model,
    generationConfig: {
      maxOutputTokens: options?.maxTokens ?? 4096,
      temperature: options?.temperature ?? 0.7,
      responseMimeType: options?.jsonMode ? "application/json" : undefined,
    },
    ...(options?.systemInstruction ? { systemInstruction: options.systemInstruction } : {}),
  });
  const result = await model.generateContent(prompt);
  const text = result.response.text();
  if (!text) throw new Error("Gemini returned empty response");
  return text;
}
```

```ts
export const LONGFORM_SYSTEM_INSTRUCTION = `You are a YouTube Shorts viral segment extractor. Given a long-form video transcript or audio, identify the segments most likely to go viral as 30-60 second shorts.

Score each candidate on four 0-100 dimensions:
- hookScore: how gripping the first 3 seconds of the segment are
- emotionalScore: emotional resonance (humor, surprise, empathy, anger, awe)
- informationDensity: value per second — how much the viewer learns/feels
- trendScore: alignment with current cultural trends and topics

For each candidate include:
- startMs, endMs (integers, end > start, duration between 30000ms and 60000ms)
- all four scores (0-100 integers)
- reason: one or two sentences explaining why the segment would perform
- titleSuggestion: a punchy <=60 char title in the transcript's language
- transcriptSnippet: the transcript text inside this segment (<=500 chars)

Return ONLY valid JSON matching this exact schema:
{
  "candidates": [
    {
      "startMs": number,
      "endMs": number,
      "hookScore": number,
      "emotionalScore": number,
      "informationDensity": number,
      "trendScore": number,
      "reason": string,
      "titleSuggestion": string,
      "transcriptSnippet": string
    }
  ]
}`;

export function buildTranscriptPrompt(params: {
  title: string | null;
  durationSeconds: number;
  targetCount: number;
  transcript: Array<{ text: string; offset: number; duration: number }>;
}): string {
  const transcriptText = params.transcript
    .map((s) => `[${formatMs(s.offset)}] ${s.text}`)
    .join("\n");

  return [
    `## Source`,
    `Title: ${params.title ?? "(unknown)"}`,
    `Duration: ${params.durationSeconds}s`,
    ``,
    `## Transcript with timestamps`,
    transcriptText,
    ``,
    `## Task`,
    `Extract exactly ${params.targetCount} high-potential shorts candidates. Each must be 30-60 seconds long. Segments must NOT overlap. Return JSON only.`,
  ].join("\n");
}

export function buildAudioPrompt(params: {
  title: string | null;
  durationSeconds: number;
  targetCount: number;
}): string {
  return [
    `## Source`,
    `Title: ${params.title ?? "(unknown)"}`,
    `Duration: ${params.durationSeconds}s`,
    ``,
    `## Task`,
    `The attached audio file is the full source. Listen to it and extract exactly ${params.targetCount} high-potential shorts candidates. Each must be 30-60 seconds long. Segments must NOT overlap. For transcriptSnippet, write a brief paraphrase of what is said in that segment. Return JSON only.`,
  ].join("\n");
}

function formatMs(ms: number): string {
  const totalSeconds = Math.floor(ms / 1000);
  const mm = Math.floor(totalSeconds / 60);
  const ss = totalSeconds % 60;
  return `${mm}:${ss.toString().padStart(2, "0")}`;
}
```

### Task 5: Segment validator

**File**: `src/lib/longform/segment-validator.ts` (NEW)

**Action**: parse Gemini JSON, clamp scores 0-100, enforce non-overlap, enforce duration bounds per segment (30-60s), and truncate/pad to the requested target count. Return strongly-typed `ValidatedCandidate[]`.

```ts
export type ValidatedCandidate = {
  startMs: number;
  endMs: number;
  hookScore: number;
  emotionalScore: number;
  informationDensity: number;
  trendScore: number;
  reason: string;
  titleSuggestion: string;
  transcriptSnippet: string;
};

const MIN_CLIP_MS = 30_000;
const MAX_CLIP_MS = 60_000;

export function parseAndValidateCandidates(
  rawJson: string,
  options: { targetCount: number; sourceDurationSeconds: number }
): ValidatedCandidate[] {
  let cleaned = rawJson.trim();
  if (cleaned.startsWith("```")) {
    cleaned = cleaned.replace(/^```(?:json)?\s*\n?/, "").replace(/\n?```\s*$/, "");
  }
  let parsed: { candidates?: unknown };
  try { parsed = JSON.parse(cleaned); } catch (err) {
    throw new Error(`Gemini JSON parse failed: ${(err as Error).message}`);
  }
  if (!Array.isArray(parsed.candidates)) {
    throw new Error("Gemini response missing 'candidates' array");
  }

  const sourceMs = options.sourceDurationSeconds * 1000;

  const validated: ValidatedCandidate[] = parsed.candidates
    .map((c) => normalizeCandidate(c as Record<string, unknown>, sourceMs))
    .filter((c): c is ValidatedCandidate => c !== null);

  // Sort by startMs
  validated.sort((a, b) => a.startMs - b.startMs);

  // Drop overlaps (keep earlier segment)
  const nonOverlapping: ValidatedCandidate[] = [];
  for (const cand of validated) {
    const last = nonOverlapping[nonOverlapping.length - 1];
    if (!last || cand.startMs >= last.endMs) nonOverlapping.push(cand);
  }

  // Truncate to target count
  return nonOverlapping.slice(0, options.targetCount);
}

function normalizeCandidate(
  raw: Record<string, unknown>,
  sourceMs: number
): ValidatedCandidate | null {
  const startMs = clampInt(raw.startMs, 0, sourceMs);
  let endMs = clampInt(raw.endMs, 0, sourceMs);
  if (endMs <= startMs) return null;
  let duration = endMs - startMs;
  if (duration < MIN_CLIP_MS) {
    endMs = Math.min(sourceMs, startMs + MIN_CLIP_MS);
    duration = endMs - startMs;
  }
  if (duration > MAX_CLIP_MS) {
    endMs = startMs + MAX_CLIP_MS;
  }
  if (endMs <= startMs) return null;

  return {
    startMs,
    endMs,
    hookScore: clampInt(raw.hookScore, 0, 100),
    emotionalScore: clampInt(raw.emotionalScore, 0, 100),
    informationDensity: clampInt(raw.informationDensity, 0, 100),
    trendScore: clampInt(raw.trendScore, 0, 100),
    reason: typeof raw.reason === "string" ? raw.reason.slice(0, 1000) : "",
    titleSuggestion: typeof raw.titleSuggestion === "string" ? raw.titleSuggestion.slice(0, 120) : "",
    transcriptSnippet: typeof raw.transcriptSnippet === "string" ? raw.transcriptSnippet.slice(0, 500) : "",
  };
}

function clampInt(value: unknown, min: number, max: number): number {
  const n = typeof value === "number" ? value : parseInt(String(value ?? 0), 10);
  if (isNaN(n)) return min;
  return Math.max(min, Math.min(max, Math.round(n)));
}
```

### Task 6: Audio extraction helper (FFmpeg spawn)

**File**: `src/lib/video/extract-audio.ts` (NEW)

**Action**: small helper that takes an input video path and produces a downsampled mono 32 kbps MP3, suitable for Gemini Files API upload (keeps cost down per D-15 risks section).

```ts
import { spawn } from "child_process";

export async function extractAudioForAnalysis(
  inputPath: string,
  outputPath: string
): Promise<void> {
  return new Promise((resolve, reject) => {
    const args = [
      "-y",
      "-i", inputPath,
      "-vn",
      "-ac", "1",
      "-ar", "16000",
      "-b:a", "32k",
      "-f", "mp3",
      outputPath,
    ];
    const proc = spawn("ffmpeg", args);
    let stderrTail = "";
    proc.stderr.on("data", (d) => { stderrTail = (stderrTail + d.toString()).slice(-4000); });
    proc.on("close", (code) => {
      if (code === 0) resolve();
      else reject(new Error(`ffmpeg audio extract failed (${code}): ${stderrTail}`));
    });
    proc.on("error", (err) => reject(err));
  });
}

### Task 7: Implement `longform-analyze` handler

**File**: `src/worker/handlers/longform-analyze.ts` (MODIFY — replace the stub)

**Action**: full handler:

1. Load `longform_sources` row by `payload.sourceId`
2. Require `source.status === 'ready'`
3. Update `longform_sources.status = 'analyzing'`
4. Determine mode:
   - If `sourceType === 'url'` and `source.sourceUrl` matches a YouTube video, try `fetchTranscript(videoId)`. On success → transcript mode. On `null` (no captions) → audio mode.
   - If `sourceType === 'file'` → audio mode.
5. `getUserAIClient(userId, 'gemini')` — fail fast if no Gemini key (audio mode requires Gemini)
6. **Transcript mode**:
   - Persist transcript on the source: `longform_sources.transcript = { segments, fullText, language, source: 'youtube-transcript' }`
   - Build prompt via `buildTranscriptPrompt`
   - Call `provider.generateTextWithModel(prompt, { systemInstruction: LONGFORM_SYSTEM_INSTRUCTION, jsonMode: true, temperature: 0.3, model: 'gemini-2.5-pro', maxTokens: 8192 })`
7. **Audio mode**:
   - Download source from Supabase to tempDir via `downloadLongformSource(storagePath)`
   - `extractAudioForAnalysis(sourcePath, audioPath)`
   - `provider.generateJsonFromAudio({ audioPath, mimeType: 'audio/mpeg', prompt, systemInstruction, model: 'gemini-2.5-pro' })`
   - If the provider lacks the method (OpenAI), throw with a descriptive error.
8. `parseAndValidateCandidates(response, { targetCount, sourceDurationSeconds: source.durationSeconds })`
9. Insert all candidates into `longform_candidates` in one `db.insert(...).values([...])`
10. Set `longform_sources.status='ready'` (analyzed) — or add a new status `analyzed`. Use `analyzed` for clarity.
11. Update `jobs.status='completed'`, `result={ candidateCount }`
12. `finally`: cleanup tempDir

`payload` shape:
```ts
{ sourceId: string; targetCount?: number; mode?: 'auto' | 'transcript' | 'audio' }
```
Default `targetCount = 8`, `mode = 'auto'`.

Mirror the progress/event pattern from `longform-download.ts` and `export-video.ts`. Skeleton:

```ts
import type { Job } from "bullmq";
import { eq } from "drizzle-orm";
import { mkdtemp, rm, writeFile } from "fs/promises";
import { join } from "path";
import { tmpdir } from "os";
import { jobs, jobEvents, longformSources, longformCandidates } from "@/lib/db/schema";
import { getUserAIClient } from "@/lib/ai/get-user-ai-client";
import { fetchTranscript } from "@/lib/youtube/transcript";
import { parseVideoUrl } from "@/lib/youtube/parse-url";
import {
  LONGFORM_SYSTEM_INSTRUCTION,
  buildTranscriptPrompt,
  buildAudioPrompt,
} from "@/lib/longform/analyze-prompt";
import { parseAndValidateCandidates } from "@/lib/longform/segment-validator";
import { downloadLongformSource } from "@/lib/media/longform-storage";
import { extractAudioForAnalysis } from "@/lib/video/extract-audio";

// ... same DrizzleInstance type as other handlers
export async function handleLongformAnalyze(job: Job, db: DrizzleInstance) {
  const { jobId, userId, payload } = job.data as {
    jobId: string; userId: string;
    payload: { sourceId: string; targetCount?: number; mode?: "auto" | "transcript" | "audio" };
  };
  const sourceId = payload.sourceId;
  const targetCount = Math.max(5, Math.min(10, payload.targetCount ?? 8));
  let tempDir: string | null = null;

  try {
    await db.update(jobs).set({ status: "active", currentStep: "loading source", updatedAt: new Date() }).where(eq(jobs.id, jobId));
    await db.insert(jobEvents).values({ jobId, event: "started", data: { sourceId, targetCount } });

    const [source] = await db.select().from(longformSources).where(eq(longformSources.id, sourceId));
    if (!source) throw new Error(`longform_source ${sourceId} not found`);
    if (source.status !== "ready") throw new Error(`source not ready: status=${source.status}`);

    await db.update(longformSources).set({ status: "analyzing", updatedAt: new Date() }).where(eq(longformSources.id, sourceId));

    const { provider } = await getUserAIClient(userId, "gemini");
    if (!provider.generateTextWithModel || !provider.generateJsonFromAudio) {
      throw new Error("Gemini provider missing longform methods — update src/lib/ai/gemini.ts");
    }

    // Choose mode
    let chosenMode: "transcript" | "audio" = "audio";
    let transcript: Awaited<ReturnType<typeof fetchTranscript>> | null = null;

    if (payload.mode !== "audio" && source.sourceType === "url" && source.sourceUrl) {
      const parsed = parseVideoUrl(source.sourceUrl);
      if (parsed) {
        await db.update(jobs).set({ currentStep: "fetching transcript", progress: 10, updatedAt: new Date() }).where(eq(jobs.id, jobId));
        transcript = await fetchTranscript(parsed.videoId);
        if (transcript) chosenMode = "transcript";
      }
    }

    let rawJson: string;

    if (chosenMode === "transcript" && transcript) {
      await db.update(longformSources).set({
        transcript: { ...transcript, source: "youtube-transcript" },
        updatedAt: new Date(),
      }).where(eq(longformSources.id, sourceId));

      const prompt = buildTranscriptPrompt({
        title: source.title,
        durationSeconds: source.durationSeconds ?? 0,
        targetCount,
        transcript: transcript.segments,
      });

      await db.update(jobs).set({ currentStep: "analyzing transcript with gemini-2.5-pro", progress: 40, updatedAt: new Date() }).where(eq(jobs.id, jobId));
      rawJson = await provider.generateTextWithModel(prompt, {
        systemInstruction: LONGFORM_SYSTEM_INSTRUCTION,
        jsonMode: true,
        temperature: 0.3,
        maxTokens: 8192,
        model: "gemini-2.5-pro",
      });
    } else {
      if (!source.storagePath) throw new Error("audio mode requires storagePath");

      tempDir = await mkdtemp(join(tmpdir(), "longform-analyze-"));
      const videoPath = join(tempDir, "source.mp4");
      const audioPath = join(tempDir, "audio.mp3");

      await db.update(jobs).set({ currentStep: "downloading source", progress: 10, updatedAt: new Date() }).where(eq(jobs.id, jobId));
      const buffer = await downloadLongformSource(source.storagePath);
      await writeFile(videoPath, buffer);

      await db.update(jobs).set({ currentStep: "extracting audio", progress: 30, updatedAt: new Date() }).where(eq(jobs.id, jobId));
      await extractAudioForAnalysis(videoPath, audioPath);

      const prompt = buildAudioPrompt({
        title: source.title,
        durationSeconds: source.durationSeconds ?? 0,
        targetCount,
      });

      await db.update(jobs).set({ currentStep: "uploading audio to gemini", progress: 50, updatedAt: new Date() }).where(eq(jobs.id, jobId));
      rawJson = await provider.generateJsonFromAudio({
        audioPath,
        mimeType: "audio/mpeg",
        prompt,
        systemInstruction: LONGFORM_SYSTEM_INSTRUCTION,
        model: "gemini-2.5-pro",
      });
    }

    await db.update(jobs).set({ currentStep: "parsing candidates", progress: 85, updatedAt: new Date() }).where(eq(jobs.id, jobId));
    const candidates = parseAndValidateCandidates(rawJson, {
      targetCount,
      sourceDurationSeconds: source.durationSeconds ?? 0,
    });
    if (candidates.length === 0) throw new Error("Gemini returned zero valid candidates");

    await db.insert(longformCandidates).values(
      candidates.map((c) => ({
        sourceId,
        startMs: c.startMs,
        endMs: c.endMs,
        hookScore: c.hookScore,
        emotionalScore: c.emotionalScore,
        informationDensity: c.informationDensity,
        trendScore: c.trendScore,
        reason: c.reason,
        titleSuggestion: c.titleSuggestion,
        transcriptSnippet: c.transcriptSnippet,
      }))
    );

    await db.update(longformSources).set({ status: "analyzed", updatedAt: new Date() }).where(eq(longformSources.id, sourceId));
    await db.update(jobs).set({
      status: "completed",
      progress: 100,
      currentStep: "analysis complete",
      result: { sourceId, candidateCount: candidates.length, mode: chosenMode },
      updatedAt: new Date(),
    }).where(eq(jobs.id, jobId));
    await db.insert(jobEvents).values({ jobId, event: "completed", data: { sourceId, candidateCount: candidates.length } });

    return { sourceId, candidateCount: candidates.length };
  } catch (error) {
    const errorMessage = error instanceof Error ? error.message : String(error);
    await db.update(jobs).set({ status: "failed", errorMessage, updatedAt: new Date() }).where(eq(jobs.id, jobId));
    await db.update(longformSources).set({ status: "failed", errorMessage, updatedAt: new Date() }).where(eq(longformSources.id, sourceId));
    await db.insert(jobEvents).values({ jobId, event: "failed", data: { error: errorMessage } });
    throw error;
  } finally {
    if (tempDir) await rm(tempDir, { recursive: true, force: true });
  }
}
```

### Task 8: Add `POST /api/longform/sources/[id]/analyze` route

**File**: `src/app/api/longform/sources/[id]/analyze/route.ts` (NEW)

**Action**: endpoint the UI calls after download completes. Creates a `jobs` row and enqueues `longform-analyze` on `getLongformQueue()`.

```ts
import { NextRequest, NextResponse } from "next/server";
import { z } from "zod";
import { and, eq } from "drizzle-orm";
import { db } from "@/lib/db";
import { jobs, longformSources } from "@/lib/db/schema";
import { getLongformQueue } from "@/lib/queue-longform";
import { getServerSession } from "@/lib/auth/get-session";

const bodySchema = z.object({
  targetCount: z.number().int().min(5).max(10).optional(),
  mode: z.enum(["auto", "transcript", "audio"]).optional(),
});

export async function POST(request: NextRequest, { params }: { params: Promise<{ id: string }> }) {
  const session = await getServerSession();
  if (!session?.user) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  const { id } = await params;
  const body = bodySchema.safeParse(await request.json().catch(() => ({})));
  const payloadIn = body.success ? body.data : {};

  const [source] = await db.select().from(longformSources).where(and(eq(longformSources.id, id), eq(longformSources.userId, session.user.id)));
  if (!source) return NextResponse.json({ error: "Not found" }, { status: 404 });
  if (source.status !== "ready") return NextResponse.json({ error: `source not ready (status=${source.status})` }, { status: 409 });

  const [jobRow] = await db.insert(jobs).values({
    userId: session.user.id,
    type: "longform-analyze",
    status: "pending",
    progress: 0,
    payload: { sourceId: id, targetCount: payloadIn.targetCount, mode: payloadIn.mode },
  }).returning();

  await getLongformQueue().add("longform-analyze", {
    payload: { sourceId: id, targetCount: payloadIn.targetCount, mode: payloadIn.mode },
    jobId: jobRow.id,
    userId: session.user.id,
  });

  return NextResponse.json({ jobId: jobRow.id }, { status: 201 });
}
```

## Risks & Mitigations

- **Risk**: Gemini 2.5 Pro 1M-token context may not fit a 4-hour transcript (unlikely but possible). **Mitigation**: detected at runtime via error handling; future fallback would be sliding window. Noted in CONTEXT.md as medium risk.
- **Risk**: Gemini Files API costs grow with audio duration. **Mitigation**: Task 6 extracts 16 kHz mono 32 kbps mp3, which keeps 1 hour of audio under ~15 MB.
- **Risk**: YouTube captions contain inaccurate timestamps, causing off-by-seconds clips. **Mitigation**: segment validator clamps bounds to `0..sourceDurationMs`; off-by-a-few-seconds is acceptable for MVP.
